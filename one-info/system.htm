Vics One app TechTool 6- Create Restore Point

Important Scripting Note Windows limits automatic restore point creation to once every 24 hours by default. If your script runs create restore multiple times a day, subsequent attempts will be ignored unless you modify the registry throttle using this command: Set-ItemProperty -Path "HKLM:\SOFTWARE\Microsoft\Windows NT\CurrentVersion\SystemRestore"
-Name "SystemRestorePointCreationFrequency" -Value 0

Vics One app TechTool 7- BackUp Registry


Back Up Registry:
   # --- CONFIGURATION ---
$Timestamp = Get-Date -Format "MM_dd_yyyy hh.mm tt"
$SystemDrive = $env:SystemDrive
$BackupRoot = "$SystemDrive\RegistryBackUp\Registry $Timestamp"
$TempLink = "$SystemDrive\ShadowLink"

if (!(Test-Path $BackupRoot)) { New-Item -ItemType Directory -Path $BackupRoot -Force }

Write-Host "[1/4] Creating Shadow Copy..." -ForegroundColor Cyan
$shadow = Invoke-CimMethod -MethodName Create -ClassName Win32_ShadowCopy -Arguments @{Volume="$SystemDrive\"}
$shadowID = $shadow.ShadowID
$shadowPath = (Get-CimInstance -ClassName Win32_ShadowCopy | Where-Object {$_.ID -eq $shadowID}).DeviceObject + "\"

Write-Host "[2/4] Mounting Shadow Copy..." -ForegroundColor Cyan
cmd /c mklink /d "$TempLink" "$shadowPath"

Write-Host "[3/4] Copying System & User Hives..." -ForegroundColor Cyan
# Copy System Hives
$sysDest = "$BackupRoot\System32\config"
robocopy "$TempLink\Windows\System32\config" "$sysDest" SOFTWARE SYSTEM SAM SECURITY DEFAULT /R:1 /W:1 /MT:16

# Copy All User Hives (NTUSER.DAT and UsrClass.dat)
$userProfiles = Get-ChildItem "$TempLink\Users" -Directory
foreach ($profile in $userProfiles) {
    $username = $profile.Name
    $userDest = "$BackupRoot\Users\$username"
    
    # NTUSER.DAT
    robocopy "$TempLink\Users\$username" "$userDest" NTUSER.DAT /B /R:1 /W:1 /MT:16
    
    # UsrClass.dat
    $usrClassPath = "$TempLink\Users\$username\AppData\Local\Microsoft\Windows"
    if (Test-Path "$usrClassPath\UsrClass.dat") {
        robocopy "$usrClassPath" "$userDest\AppData\Local\Microsoft\Windows" UsrClass.dat /B /R:1 /W:1 /MT:16
    }
}

Write-Host "[4/4] Cleaning up..." -ForegroundColor Cyan
if (Test-Path $TempLink) { cmd /c rmdir "$TempLink" }
Get-CimInstance -ClassName Win32_ShadowCopy | Where-Object {$_.ID -eq $shadowID} | Remove-CimInstance

Write-Host "`nBackup Complete! Saved to: $BackupRoot" -ForegroundColor Green
